var ErrEmptyFile = errors.New("the string is empty")

func CountWords(content string) (int, error) {
	if content == "" {
		return 0, ErrEmptyFile
	} // checking if the string is empty or not

	words := strings.Fields(content)
	// Fields splits the string s around each instance of one or more consecutive white space characters, as defined by unicode.IsSpace, returning a slice of substrings.

	return len(words), nil

}

func CreateFile(name string) (*os.File, error) {
	f, err := os.Create(name)
	if err != nil {
		return f, err
	}
	return f, nil
}
